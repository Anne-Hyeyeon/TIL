## 트리 만들기
```js
const arrayToTree = (arr) => {
  let tempResult = {},
    roots = []
    
  const getTreeFunc = () => {
      arr.forEach((el, i) => {
        tempResult[el.dirId] = i;
        el.children = [];
        if (el.dirPid !== -1) {
          arr[tempResult[el.dirPid]].children.push(el);
        } else {
          roots.push(el);
        };
      });
    }
  getTreeFunc();
  return roots;
};

arrayToTree(data);
```
